#if TMPL_COM_HAS_CONSOLE == 0
int main(void)
#else
int main(int argc, char *argv[])
#endif
{
uint8_t ubNetV = eCOM_NET_1;
//---------------------------------------------------------------------------------------------------
// Print application information
//
ComDemoAppConsoleLog(" ");
ComDemoAppConsoleLog("###############################################################################");
ComDemoAppConsoleLog("# CANopen FD Master Example #");
ComDemoAppConsoleLog("###############################################################################");
//---------------------------------------------------------------------------------------------------
// Set options for the CANopen (FD) Master protocol stack to default values
//
ComDemoAppOptionsInit(ubNetV);
//---------------------------------------------------------------------------------------------------
// A console based application can modify the default options via parameters
//
#if TMPL_COM_HAS_CONSOLE > 0
if (ComDemoAppOptionsParse(ubNetV, argc, argv) == false)
{
ComDemoAppExit();
}
ComDemoAppConsoleLog("Use CTRL-C to quit this demo.");
ComDemoAppConsoleLog(" ");
#endif
//---------------------------------------------------------------------------------------------------
// Initialise the CANopen (FD) Master demo application
//
ComDemoAppInit(ubNetV);
//---------------------------------------------------------------------------------------------------
// this is the main loop of the embedded application
//
while (ComDemoAppRun() == true)
{
ComMgrProcess(ubNetV);
ComDemoAppProcess(ubNetV);
//-------------------------------------------------------------------------------------------
// The following code is for automatic testing only
//
#if TMPL_COM_TEST_MAIN > 0
static uint32_t ulTestMainCounterS = TMPL_COM_TEST_MAIN;
if (ulTestMainCounterS == 0) break;
ulTestMainCounterS--;
#endif
}
ComDemoAppRelease(ubNetV);
}
