static McFifoCtrl_ts atsNmtFifoS[COM_NET_MAX];
static uint16_t atsNmtDataS[COM_NET_MAX][DEMO_NMT_FIFO_SIZE];

//--------------------------------------------------------------------------------------------------------------------//
// ComDemoNmtConfigurationDone() //
// //
//--------------------------------------------------------------------------------------------------------------------//
void ComDemoNmtConfigurationDone(uint8_t ubNetV, uint8_t ubNodeIdV)
{
(void) ubNodeIdV;
atsNmtNodeS[ubNetV] = 0;
}
//--------------------------------------------------------------------------------------------------------------------//
// ComDemoNmtInit() //
// //
//--------------------------------------------------------------------------------------------------------------------//
ComStatus_tv ComDemoNmtInit(uint8_t ubNetV)
{
MC_FIFO_INIT(atsNmtFifoS[ubNetV], DEMO_NMT_FIFO_SIZE);
atsNmtNodeS[ubNetV] = 0;
return (eCOM_ERR_NONE);
}
//--------------------------------------------------------------------------------------------------------------------//
// ComDemoNmtProcess() //
// //
//--------------------------------------------------------------------------------------------------------------------//
ComStatus_tv ComDemoNmtProcess(uint8_t ubNetV)
{
uint16_t uwFifoValueT = 0;
if (atsNmtNodeS[ubNetV] == 0)
{
if (MC_FIFO_EMPTY(atsNmtFifoS[ubNetV]) == 0)
{
uwFifoValueT = atsNmtDataS[ubNetV][MC_FIFO_DATA_OUT(atsNmtFifoS[ubNetV])];
MC_FIFO_INC_OUT(atsNmtFifoS[ubNetV]);
uint8_t ubNodeIdT = (uint8_t) uwFifoValueT;
uint8_t ubNmtStateT = (uint8_t) (uwFifoValueT >> 8);
switch (ubNmtStateT)
{
case eCOM_NMT_STATE_BOOTUP:
atsNmtNodeS[ubNetV] = ubNodeIdT;
ComDemoNodeIdentify(ubNetV, ubNodeIdT);
break;
case eCOM_NMT_STATE_PREOPERATIONAL:
ComNmtSetNodeState(ubNetV, ubNodeIdT, eCOM_NMT_STATE_OPERATIONAL);
break;
case eCOM_NMT_STATE_OPERATIONAL:
break;
default:
break;
}
}
}
return (eCOM_ERR_NONE);
}
//--------------------------------------------------------------------------------------------------------------------//
// ComDemoNmtStateChange() //
// //
//--------------------------------------------------------------------------------------------------------------------//
void ComDemoNmtStateChange(uint8_t ubNetV, uint8_t ubNodeIdV, uint8_t ubNmtStateV)
{
uint16_t uwFifoValueT = 0;
uwFifoValueT = ubNmtStateV;
uwFifoValueT = uwFifoValueT << 8;
uwFifoValueT = uwFifoValueT + ubNodeIdV;
if (MC_FIFO_FULL(atsNmtFifoS[ubNetV]) == 0)
{
//------------------------------------------------------------------------------------------------
// Push NMT state and node-id to FIFO
//
atsNmtDataS[ubNetV][MC_FIFO_DATA_IN(atsNmtFifoS[ubNetV])] = uwFifoValueT;
MC_FIFO_INC_IN(atsNmtFifoS[ubNetV]);
}
}
//--------------------------------------------------------------------------------------------------------------------//
// DemoComNmtStartNet() //
// Start all nodes in the network //
//--------------------------------------------------------------------------------------------------------------------//
ComStatus_tv DemoComNmtStartNodes(uint8_t ubNetV)
{
ComStatus_tv tvStatusT;
//---------------------------------------------------------------------------------------------------
// set NMT inhibit time to 500 ms
//
tvStatusT = ComNmtSetInhibit(eCOM_NET_1, 500000);
if(tvStatusT != eCOM_ERR_NONE)
{
return(tvStatusT);
}
//---------------------------------------------------------------------------------------------------
// send "Reset all Nodes" command
//
ComNmtSetNodeState(ubNetV, 0, eCOM_NMT_STATE_RESET_NODE);
//---------------------------------------------------------------------------------------------------
// now send "Operational" command, wait until inhibit time is over
//
do {
tvStatusT = ComNmtSetNodeState( ubNetV, 0, eCOM_NMT_STATE_OPERATIONAL);
} while (tvStatusT == eCOM_ERR_SERVICE_MODE);
return (tvStatusT);
} // DemoComNmtStartNodes

// DemoComNmtPrintNodeState() //
// Print node state in text buffer //
//--------------------------------------------------------------------------------------------------------------------//
#if COM_DEMO_PRINT_SUPPORT > 0
ComStatus_tv DemoComNmtPrintNodeState( uint8_t ubNetV, uint8_t ubNodeIdV, char * szBufferV)
{
uint8_t ubNodeStateT;
ComStatus_tv tvStatusT;
tvStatusT = ComNmtGetNodeState(ubNetV, ubNodeIdV, &ubNodeStateT);
if(tvStatusT != eCOM_ERR_NONE)
{
sprintf(szBufferV, "COM-ERR: %d", (int) tvStatusT);
return(tvStatusT);
}
switch(ubNodeStateT)
{
case eCOM_NMT_STATE_STOPPED:
sprintf(szBufferV, "NID %d: %s", ubNodeIdV, "Stopped ");
break;
case eCOM_NMT_STATE_PREOPERATIONAL:
sprintf(szBufferV, "NID %d: %s", ubNodeIdV, "Pre-Op. ");
break;
case eCOM_NMT_STATE_OPERATIONAL:
sprintf(szBufferV, "NID %d: %s", ubNodeIdV, "Operational ");
break;
case eCOM_NMT_STATE_RESET_NODE:
case eCOM_NMT_STATE_RESET_COM:
sprintf(szBufferV, "NID %d: %s", ubNodeIdV, "Reset ");
break;
case eCOM_NMT_STATE_BOOTUP:
sprintf(szBufferV, "NID %d: %s", ubNodeIdV, "Booted ");
break;
default:
sprintf(szBufferV, "NID %d: %s", ubNodeIdV, "Unknown ");
break;
}
return(eCOM_ERR_NONE);
} // DemoComNmtPrintNodeState
#endif
