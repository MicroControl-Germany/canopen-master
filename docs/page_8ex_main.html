//------------------------------------------------------------------------------------------------------

#include "com_mgr.h"
#include "com_nmt.h"
//------------------------------------------------------------------------------------------------------
// HAL header files

#include "mc_cpu.h"
#include "mc_tmr.h"

//------------------------------------------------------------------------------------------------------
// CANopen Master demo application
//
#include "com_demo_app.h"
/*--------------------------------------------------------------------------------------------------------------------*\
** Definitions **
** **
\*--------------------------------------------------------------------------------------------------------------------*/
//------------------------------------------------------------------------------------------------------
// Specific configuration of CANopen Master Protocol Stack
//
#ifndef TMPL_COM_CONFIG
#define TMPL_COM_CONFIG eCOM_MODE_NMT_MASTER
#endif
//------------------------------------------------------------------------------------------------------
// Run CANopen Master Protocol Stack for internal testing if > 0
//
#ifndef TMPL_COM_TEST_MAIN
#define TMPL_COM_TEST_MAIN 0
#endif
//------------------------------------------------------------------------------------------------------
// Select physical CAN interface for CANopen Master Protocol Stack
//
#ifndef TMPL_COM_USE_CAN_IF
#define TMPL_COM_USE_CAN_IF eCP_CHANNEL_1
#endif
//--------------------------------------------------------------------------------------------------------------------//
// main() //
// //
//--------------------------------------------------------------------------------------------------------------------//
int main(void)
{
//---------------------------------------------------------------------------------------------------
// Initialise the target CPU
//
McCpuInit();
//---------------------------------------------------------------------------------------------------
// Initialise the timer resource on the target CPU
//
McTmrInit();
//---------------------------------------------------------------------------------------------------
// Initialise the CANopen Master stack with address 127, bit-rate = 500K
//
ComMgrInit( TMPL_COM_USE_CAN_IF, eCOM_NET_1, eCP_BITRATE_500K, 127, TMPL_COM_CONFIG);
//---------------------------------------------------------------------------------------------------
// Initialise the CANopen Master demo application, this is individual for every target
//
ComDemoAppInit(eCOM_NET_1);
//---------------------------------------------------------------------------------------------------
// Start the CANopen Master stack cyclic timer event function on a 1 ms base
//
McTmrFunctionInit(ComMgrTimerEvent, McTmrTimeToTicks(COM_TIMER_PERIOD), eTMR_CTRL_START);
//---------------------------------------------------------------------------------------------------
// start the CANopen master stack:
// this will issue the boot-up message of the CANopen master
//
ComMgrStart(eCOM_NET_1);
//---------------------------------------------------------------------------------------------------
// start master detection procedure
//
ComNmtMasterDetection(eCOM_NET_1, 1);
//---------------------------------------------------------------------------------------------------
// Set Heartbeart producer time of 1000 ms for CANopen Master
//
ComNmtSetHbProdTime(eCOM_NET_1, 1000);
//---------------------------------------------------------------------------------------------------
// Start the demo application cyclic timer event function on a 10 ms base
//
McTmrFunctionInit(ComDemoAppTimerEvent, McTmrTimeToTicks(10000), eTMR_CTRL_START);
//---------------------------------------------------------------------------------------------------
// this is the main loop of the embedded application
//
while (1)
{
ComMgrProcess(eCOM_NET_1);
ComDemoAppProcess();
//-------------------------------------------------------------------------------------------
// The following code is for automatic testing only
//
#if TMPL_COM_TEST_MAIN > 0
static uint32_t ulTestMainCounterS = TMPL_COM_TEST_MAIN;
if (ulTestMainCounterS == 0) break;
ulTestMainCounterS--;
#endif
}
}
